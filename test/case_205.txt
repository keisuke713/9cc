typedef struct Person Person;

struct Person {
    char *name;
    int len;
};

int strncmp(char *n1, char *n2, int len) {
    int i;
    for (i = 0; i < len; i++) {
        if (n1[i] < n2[i])
            return 1;
        if (n1[i] > n2[i])
            return -1;
    }
    return 0;
}

int main() {
    Person p1;
    p1.name = "keisuke";
    p1.len = 7;

    Person p2;
    p2.name = "keisuke";
    p2.len = 7;

    strncmp(p1.name, p2.name, p1.len);
}
